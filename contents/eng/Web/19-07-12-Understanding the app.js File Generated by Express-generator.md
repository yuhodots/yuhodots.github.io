When you look at the app.js file of a web application project generated by Express-generator, you'll find about 40 lines of code. Today, let's study what each of these lines means. First, here is the source code of the app.js file.

```javascript
var createError = require('http-errors');
var express = require('express');
var path = require('path');
var cookieParser = require('cookie-parser');
var logger = require('morgan');

var indexRouter = require('./routes/index');
var usersRouter = require('./routes/users');

var app = express();

// view engine setup
app.set('views', path.join(__dirname, 'views'));
app.set('view engine', 'jade');

app.use(logger('dev'));
app.use(express.json());
app.use(express.urlencoded({ extended: false }));
app.use(cookieParser());
app.use(express.static(path.join(__dirname, 'public')));

app.use('/', indexRouter);
app.use('/users', usersRouter);

// catch 404 and forward to error handler
app.use(function(req, res, next) {
  next(createError(404));
});

// error handler
app.use(function(err, req, res, next) {
  // set locals, only providing error in development
  res.locals.message = err.message;
  res.locals.error = req.app.get('env') === 'development' ? err : {};

  // render the error page
  res.status(err.status || 500);
  res.render('error');
});

module.exports = app;
```

Let's break the code above into three major parts: **module declarations, server configuration, and middleware registration**.

### Module Declarations

- `var createError = require('http-errors')`: An error handling [module](https://www.npmjs.com/package/http-errors)
- `var path = require('path')`: A path-related [module](http://chromeextension.kr/2016/08/node-js-path/). For example, `path.join(path1, path2...)` concatenates the provided paths into a single path.
- `var cookieParser = require('cookie-parser')`: A module that adds the `cookies` property and `cookie()` method to the req and res objects.
- `var logger = require('morgan')`: A logging [module](https://www.npmjs.com/package/morgan). For example, with `app.use(logger('dev'))`, access attempts to the server will be logged.
- `var indexRouter = require('./routes/index')`: A module for handling HTTP GET / requests.
- `var usersRouter = require('./routes/users')`: A module for handling HTTP GET /users requests.

### Configuration

\_\_dirname refers to **the location of the current file**, i.e., the location of app.js. So the result of `path.join(__dirname, 'views')` is the path **'current directory/views'**.

- `app.set('views', path.join(__dirname, 'views'))`: Sets the default views folder to `path.join(__dirname, 'views')`
- `app.set('view engine', 'jade')`: Sets the default template engine to jade

### Middleware Registration

- `app.use(logger('dev'))`: Logs access attempts to the server
- `app.use(express.json())`: Parses JSON-formatted HTTP request bodies
- `app.use(express.urlencoded({ extended: false }))`: Encodes non-English (e.g., Korean) URLs
- `app.use(cookieParser())`: Adds the `cookies` property and `cookie()` method to req and res objects
- `app.use(express.static(path.join(__dirname, 'public')))`: Serves the contents of `path.join(__dirname, 'public')` as static files from the web server root
- `app.use('/', indexRouter)`: Handles HTTP GET / requests
- `app.use('/users', usersRouter)`: Handles HTTP GET /users requests

The remaining code at the bottom handles error management.
